SELECT IL.NAME AS TASKLIST, I.SUMMARY AS TASK from ISSUESLISTS IL
JOIN ISSUES I ON IL.ID = I.ISSUESLIST_ID;
SELECT I.SUMMARY AS TASK, U.FIRSTNAME, U.LASTNAME from ISSUES I 
JOIN USERS U ON I.USER_ID_ASSIGNEDTO = U.ID;
SELECT I.SUMMARY, COUNT(*) AS NUMBER_OF_TASK, U.FIRSTNAME, U.LASTNAME from ISSUES I
JOIN USERS U ON I.USER_ID_ASSIGNEDTO = U.ID
GROUP BY U.ID;
